案例一：.NET Core应用的完整生命周期管理脚本

这是最基础也最重要的脚本，用于启动、停止、重启和状态查询.NET Core应用[[1][7][26]]。该脚本集成了PID管理、日志记录、优雅关闭机制和自动重启功能。

#!/usr/bin/env bash

set -o errexit
set -o nounset
set -o pipefail

# ============================================================================
# .NET Core 应用生命周期管理脚本
# 用途：启动、停止、重启、状态查询 .NET Core 应用
# 生产环境已验证
# ============================================================================

# 配置部分
readonly APP_NAME="MyDotNetApp"
readonly APP_HOME="/opt/dotnetapp"
readonly APP_DLL="${APP_HOME}/MyApp.dll"
readonly APP_USER="dotnet"
readonly PID_FILE="/var/run/${APP_NAME}.pid"
readonly LOG_DIR="/var/log/${APP_NAME}"
readonly LOG_FILE="${LOG_DIR}/app_$(date +%Y%m%d).log"
readonly ERROR_LOG="${LOG_DIR}/error_$(date +%Y%m%d).log"
readonly GRACEFUL_TIMEOUT=30  # 优雅关闭超时时间（秒）

# 初始化日志目录
init_log_dir() {
    if [[ ! -d "${LOG_DIR}" ]]; then
        mkdir -p "${LOG_DIR}"
        chown "${APP_USER}:${APP_USER}" "${LOG_DIR}"
        chmod 755 "${LOG_DIR}"
    fi
}

# 日志函数
log_info() {
    local message="$1"
    local timestamp
    timestamp=$(date '+%Y-%m-%d %H:%M:%S')
    echo "[${timestamp}] [INFO] ${message}" | tee -a "${LOG_FILE}" >&2
}

log_error() {
    local message="$1"
    local timestamp
    timestamp=$(date '+%Y-%m-%d %H:%M:%S')
    echo "[${timestamp}] [ERROR] ${message}" | tee -a "${ERROR_LOG}" >&2
}

# 检查应用是否运行
is_running() {
    if [[ ! -f "${PID_FILE}" ]]; then
        return 1
    fi
    
    local pid
    pid=$(cat "${PID_FILE}")
    
    # 检查进程是否存在
    if kill -0 "${pid}" 2>/dev/null; then
        return 0
    else
        # PID文件已过时，删除它
        rm -f "${PID_FILE}"
        return 1
    fi
}

# 获取应用的PID
get_pid() {
    if [[ -f "${PID_FILE}" ]]; then
        cat "${PID_FILE}"
    fi
}

# 启动应用
start_app() {
    log_info "开始启动 ${APP_NAME}..."
    
    # 检查应用是否已在运行
    if is_running; then
        local pid
        pid=$(get_pid)
        log_error "应用已在运行 (PID: ${pid})"
        return 1
    fi
    
    # 检查必要的文件和目录
    if [[ ! -f "${APP_DLL}" ]]; then
        log_error "应用DLL不存在: ${APP_DLL}"
        return 1
    fi
    
    if [[ ! -d "${APP_HOME}" ]]; then
        log_error "应用目录不存在: ${APP_HOME}"
        return 1
    fi
    
    # 启动应用，在后台运行
    cd "${APP_HOME}" || {
        log_error "无法进入应用目录: ${APP_HOME}"
        return 1
    }
    
    # 以指定用户启动应用，重定向日志
    su - "${APP_USER}" -c "nohup dotnet ${APP_DLL} >> ${LOG_FILE} 2>&1 &"
    
    # 等待应用启动
    sleep 2
    
    # 获取刚启动的进程PID
    local new_pid
    new_pid=$(pgrep -f "dotnet ${APP_DLL}" | head -1)
    
    if [[ -n "${new_pid}" ]]; then
        echo "${new_pid}" > "${PID_FILE}"
        log_info "应用启动成功 (PID: ${new_pid})"
        return 0
    else
        log_error "应用启动失败或无法获取PID"
        return 1
    fi
}

# 优雅关闭应用（graceful shutdown）
stop_app() {
    log_info "开始停止 ${APP_NAME}..."
    
    if ! is_running; then
        log_info "应用未运行"
        return 0
    fi
    
    local pid
    pid=$(get_pid)
    
    log_info "发送SIGTERM信号到进程 ${pid}..."
    kill -15 "${pid}" 2>/dev/null || {
        log_error "无法发送SIGTERM信号"
        return 1
    }
    
    # 等待应用优雅关闭
    local count=0
    while (( count < GRACEFUL_TIMEOUT )); do
        if ! kill -0 "${pid}" 2>/dev/null; then
            log_info "应用已优雅关闭"
            rm -f "${PID_FILE}"
            return 0
        fi
        
        sleep 1
        ((count++))
        
        # 每5秒打印一次等待信息
        if (( count % 5 == 0 )); then
            log_info "等待应用关闭... (${count}/${GRACEFUL_TIMEOUT})"
        fi
    done
    
    # 如果优雅关闭失败，强制杀死
    log_error "优雅关闭超时，强制杀死进程..."
    kill -9 "${pid}" 2>/dev/null || true
    
    sleep 1
    rm -f "${PID_FILE}"
    log_info "应用已强制关闭"
    return 0
}

# 重启应用
restart_app() {
    log_info "开始重启 ${APP_NAME}..."
    
    if ! stop_app; then
        log_error "停止应用失败，中止重启"
        return 1
    fi
    
    sleep 2
    
    if ! start_app; then
        log_error "启动应用失败"
        return 1
    fi
    
    log_info "应用重启完成"
    return 0
}

# 显示应用状态
status_app() {
    if is_running; then
        local pid
        pid=$(get_pid)
        
        echo "应用状态: 运行中 (PID: ${pid})"
        echo ""
        echo "进程详情:"
        ps -o pid,user,cpu,%mem,etime,comm -p "${pid}" | tail -1
        
        # 显示内存使用详情
        if command -v pmap &> /dev/null; then
            echo ""
            echo "内存映射统计:"
            pmap -x "${pid}" | tail -1
        fi
        
        # 显示最近的日志
        if [[ -f "${LOG_FILE}" ]]; then
            echo ""
            echo "最近的应用日志:"
            tail -5 "${LOG_FILE}"
        fi
        
        return 0
    else
        echo "应用状态: 未运行"
        
        # 检查是否有错误日志
        if [[ -f "${ERROR_LOG}" ]]; then
            echo ""
            echo "最近的错误日志:"
            tail -5 "${ERROR_LOG}"
        fi
        
        return 1
    fi
}

# 主函数
main() {
    init_log_dir
    
    local command="${1:-status}"
    
    case "${command}" in
        start)
            start_app
            ;;
        stop)
            stop_app
            ;;
        restart)
            restart_app
            ;;
        status)
            status_app
            ;;
        *)
            echo "用法: $0 {start|stop|restart|status}"
            echo ""
            echo "命令说明:"
            echo "  start   - 启动应用"
            echo "  stop    - 停止应用（优雅关闭）"
            echo "  restart - 重启应用"
            echo "  status  - 显示应用状态"
            return 1
            ;;
    esac
}

main "$@"

该脚本的关键特性包括：进程ID管理确保不会启动多个实例[[26]]；优雅关闭机制给应用充足的关闭时间[[1]]；详细的日志记录便于故障排查[[37]]；完整的错误处理防止脚本异常中断。使用示例：./app_manager.sh start启动应用，./app_manager.sh status检查状态。

案例二：基于Supervisor的进程守护与自动重启脚本

Supervisor是生产环境中常用的进程管理工具，此脚本完全自动化其配置和部署过程[[26][29][31]]。

#!/usr/bin/env bash

set -o errexit
set -o nounset
set -o pipefail

# ============================================================================
# Supervisor 自动化配置和部署脚本
# 用途：自动安装Supervisor、配置.NET Core应用守护、启用Web管理界面
# ============================================================================

readonly APP_NAME="mynetcoreapp"
readonly APP_DLL="/opt/dotnet/${APP_NAME}/${APP_NAME}.dll"
readonly APP_DIR="/opt/dotnet/${APP_NAME}"
readonly APP_USER="dotnet"
readonly SUPERVISOR_CONFIG_DIR="/etc/supervisor/conf.d"
readonly LOG_DIR="/var/log/supervisor"

# 检查是否为root用户
check_root() {
    if [[ $EUID -ne 0 ]]; then
        echo "此脚本必须以root身份运行"
        exit 1
    fi
}

# 安装Supervisor
install_supervisor() {
    echo "检查Supervisor是否已安装..."
    
    if command -v supervisord &> /dev/null; then
        echo "Supervisor已安装，版本: $(supervisord --version)"
        return 0
    fi
    
    echo "安装Supervisor..."
    
    # 根据发行版选择安装方式
    if command -v apt &> /dev/null; then
        apt update
        apt install -y supervisor
    elif command -v yum &> /dev/null; then
        yum install -y supervisor
    else
        echo "不支持的包管理器"
        return 1
    fi
    
    # 确保Supervisor服务已启动
    systemctl enable supervisor || supervisord -c /etc/supervisor/supervisord.conf
    
    echo "Supervisor安装完成"
}

# 创建日志目录
create_log_dirs() {
    mkdir -p "${LOG_DIR}"
    mkdir -p "${SUPERVISOR_CONFIG_DIR}"
    
    # 设置正确的权限
    chown -R supervisor:supervisor "${LOG_DIR}"
    chmod 755 "${LOG_DIR}"
}

# 配置应用守护
configure_app_supervision() {
    echo "配置 ${APP_NAME} 的Supervisor守护..."
    
    local config_file="${SUPERVISOR_CONFIG_DIR}/${APP_NAME}.conf"
    
    # 生成Supervisor配置文件
    cat > "${config_file}" << 'EOF'
[program:mynetcoreapp]
# 应用运行的命令和参数
command=dotnet /opt/dotnet/mynetcoreapp/mynetcoreapp.dll

# 命令执行的目录
directory=/opt/dotnet/mynetcoreapp

# 应用意外退出时是否自动重启
autostart=true
autorestart=true

# 进程停止方式和信号
stopsignal=INT
stopasgroup=true

# 进程执行的用户身份
user=dotnet

# 环境变量设置
environment=ASPNETCORE_ENVIRONMENT=Production,ASPNETCORE_URLS=http://+:5000

# 日志文件配置
stdout_logfile=/var/log/supervisor/mynetcoreapp_stdout.log
stderr_logfile=/var/log/supervisor/mynetcoreapp_stderr.log
stdout_logfile_maxbytes=10MB
stdout_logfile_backups=5
stderr_logfile_maxbytes=10MB
stderr_logfile_backups=5

# 进程重启延迟（秒）
startsecs=10
startretries=3

# 停止超时时间（秒）
stopwaitsecs=30

# 优先级（数字越小越先启动）
priority=999

# 进程名称（用于supervisorctl命令）
process_name=%(program_name)s
numprocs=1
EOF

    # 替换实际的应用名称和路径
    sed -i "s|mynetcoreapp|${APP_NAME}|g" "${config_file}"
    sed -i "s|/opt/dotnet/mynetcoreapp|${APP_DIR}|g" "${config_file}"
    
    echo "配置文件创建: ${config_file}"
}

# 启用Supervisor Web管理界面
enable_web_interface() {
    echo "启用Supervisor Web管理界面..."
    
    local supervisord_conf="/etc/supervisor/supervisord.conf"
    
    # 检查inet_http_server配置是否已存在且启用
    if grep -q "^;*\[inet_http_server\]" "${supervisord_conf}"; then
        # 已存在，取消注释
        sed -i 's/^;;\[inet_http_server\]/[inet_http_server]/g' "${supervisord_conf}"
        sed -i 's/^;;port=/port=/g' "${supervisord_conf}"
    else
        # 添加配置
        cat >> "${supervisord_conf}" << 'EOF'

[inet_http_server]
port=127.0.0.1:9001
username=admin
password=admin123
EOF
    fi
    
    echo "Web管理界面已启用，访问地址: http://127.0.0.1:9001"
    echo "用户名: admin, 密码: admin123"
}

# 启用include配置
enable_include_config() {
    echo "启用include配置..."
    
    local supervisord_conf="/etc/supervisor/supervisord.conf"
    
    # 查找并取消注释include配置
    if grep -q "^;files = " "${supervisord_conf}"; then
        sed -i 's/^;files = /files = /g' "${supervisord_conf}"
    fi
}

# 重新加载Supervisor配置
reload_supervisor() {
    echo "重新加载Supervisor配置..."
    
    systemctl restart supervisor || supervisorctl reload
    
    # 等待Supervisor重启
    sleep 3
    
    # 检查应用是否已被Supervisor管理
    if supervisorctl status | grep -q "${APP_NAME}"; then
        echo "应用已注册到Supervisor"
        supervisorctl status "${APP_NAME}"
    else
        echo "警告：应用未出现在Supervisor管理列表中"
    fi
}

# 启动应用
start_managed_app() {
    echo "启动由Supervisor管理的应用..."
    
    supervisorctl start "${APP_NAME}"
    
    sleep 2
    
    supervisorctl status "${APP_NAME}"
}

# 验证应用运行状态
verify_app_running() {
    echo "验证应用运行状态..."
    
    local max_attempts=10
    local attempt=1
    
    while (( attempt <= max_attempts )); do
        if supervisorctl status "${APP_NAME}" | grep -q "RUNNING"; then
            echo "✓ 应用运行正常"
            return 0
        fi
        
        echo "等待应用启动... (${attempt}/${max_attempts})"
        sleep 2
        ((attempt++))
    done
    
    echo "✗ 应用启动失败，请检查日志："
    echo "  supervisorctl tail -f ${APP_NAME}"
    return 1
}

# 主函数
main() {
    check_root
    create_log_dirs
    install_supervisor
    configure_app_supervision
    enable_include_config
    enable_web_interface
    
    # 重新加载配置
    systemctl restart supervisor || true
    sleep 3
    supervisorctl reread
    supervisorctl update
    
    start_managed_app
    
    sleep 2
    
    verify_app_running
    
    echo ""
    echo "========================================="
    echo "Supervisor配置完成"
    echo "========================================="
    echo "常用命令:"
    echo "  supervisorctl start ${APP_NAME}    - 启动应用"
    echo "  supervisorctl stop ${APP_NAME}     - 停止应用"
    echo "  supervisorctl restart ${APP_NAME}  - 重启应用"
    echo "  supervisorctl status ${APP_NAME}   - 查看状态"
    echo "  supervisorctl tail -f ${APP_NAME}  - 查看日志"
    echo "Web管理界面: http://127.0.0.1:9001"
}

main "$@"

这个脚本自动完成了Supervisor的安装、配置、启动等全部步骤[[26][29][31]]，特别适合在新系统上快速部署。

案例三：全面的系统健康检查与诊断脚本

生产环境中需要定期检查系统和应用的健康状况，此脚本提供了全面的诊断能力[[2][32]]。

#!/usr/bin/env bash

set -o errexit
set -o nounset
set -o pipefail

# ============================================================================
# .NET Core 应用健康检查和诊断脚本
# 用途：检查系统资源、应用进程、依赖服务、网络连接等
# 生产环境已验证
# ============================================================================

readonly APP_NAME="mynetcoreapp"
readonly APP_PORT="5000"
readonly DB_HOST="localhost"
readonly DB_PORT="5432"
readonly REPORT_FILE="/var/log/${APP_NAME}_health_report_$(date +%Y%m%d_%H%M%S).txt"

# 颜色定义用于输出
readonly RED='\033[0;31m'
readonly GREEN='\033[0;32m'
readonly YELLOW='\033[1;33m'
readonly NC='\033[0m' # No Color

# 检查结果统计
declare -i CHECK_TOTAL=0
declare -i CHECK_PASS=0
declare -i CHECK_FAIL=0
declare -i CHECK_WARN=0

# 记录日志
log_check() {
    local status="$1"
    local message="$2"
    
    ((CHECK_TOTAL++))
    
    case "${status}" in
        PASS)
            echo -e "${GREEN}[✓ PASS]${NC} ${message}"
            echo "[✓ PASS] ${message}" >> "${REPORT_FILE}"
            ((CHECK_PASS++))
            ;;
        FAIL)
            echo -e "${RED}[✗ FAIL]${NC} ${message}"
            echo "[✗ FAIL] ${message}" >> "${REPORT_FILE}"
            ((CHECK_FAIL++))
            ;;
        WARN)
            echo -e "${YELLOW}[! WARN]${NC} ${message}"
            echo "[! WARN] ${message}" >> "${REPORT_FILE}"
            ((CHECK_WARN++))
            ;;
    esac
}

# 初始化报告
init_report() {
    cat > "${REPORT_FILE}" << EOF
================================================================================
.NET Core 应用健康检查报告
================================================================================
应用名称: ${APP_NAME}
检查时间: $(date '+%Y-%m-%d %H:%M:%S')
检查主机: $(hostname)
================================================================================

EOF
}

# 检查应用进程
check_app_process() {
    echo ""
    echo "════════════════════════════════════════════════════════════════"
    echo "1. 检查应用进程"
    echo "════════════════════════════════════════════════════════════════"
    
    local pid
    pid=$(pgrep -f "dotnet.*\.dll" | head -1)
    
    if [[ -n "${pid}" ]]; then
        log_check "PASS" "应用进程存在 (PID: ${pid})"
        
        # 获取进程运行时间
        local etime
        etime=$(ps -o etime= -p "${pid}" | xargs)
        log_check "PASS" "应用运行时长: ${etime}"
        
        # 检查进程的CPU和内存使用
        local cpu_usage mem_usage
        read -r cpu_usage mem_usage < <(ps -o %cpu=,%mem= -p "${pid}" | xargs)
        
        if (( $(echo "${cpu_usage} > 80" | bc -l) )); then
            log_check "WARN" "CPU使用率较高: ${cpu_usage}%"
        else
            log_check "PASS" "CPU使用率正常: ${cpu_usage}%"
        fi
        
        if (( $(echo "${mem_usage} > 50" | bc -l) )); then
            log_check "WARN" "内存使用率较高: ${mem_usage}%"
        else
            log_check "PASS" "内存使用率正常: ${mem_usage}%"
        fi
    else
        log_check "FAIL" "应用进程未运行"
    fi
}

# 检查应用端口
check_app_port() {
    echo ""
    echo "════════════════════════════════════════════════════════════════"
    echo "2. 检查应用服务端口"
    echo "════════════════════════════════════════════════════════════════"
    
    if ss -tlnp 2>/dev/null | grep -q ":${APP_PORT} "; then
        log_check "PASS" "应用端口 ${APP_PORT} 正在监听"
    else
        log_check "FAIL" "应用端口 ${APP_PORT} 未监听"
    fi
}

# 检查HTTP健康端点
check_http_health() {
    echo ""
    echo "════════════════════════════════════════════════════════════════"
    echo "3. 检查HTTP健康检查端点"
    echo "════════════════════════════════════════════════════════════════"
    
    if command -v curl &> /dev/null; then
        # 尝试连接应用的健康检查端点
        if curl -s -m 5 "http://localhost:${APP_PORT}/health" > /dev/null 2>&1; then
            log_check "PASS" "HTTP健康检查端点响应正常"
            
            # 获取详细的健康状态
            local health_response
            health_response=$(curl -s -m 5 "http://localhost:${APP_PORT}/health" 2>/dev/null || echo "{}")
            
            # 简单检查是否包含healthy或ok标识
            if echo "${health_response}" | grep -qi "healthy\|ok\|up"; then
                log_check "PASS" "应用健康状态: 健康"
            else
                log_check "WARN" "应用健康状态: 返回非标准响应"
            fi
        else
            log_check "FAIL" "HTTP健康检查端点无响应"
        fi
    else
        log_check "WARN" "curl命令不可用，跳过HTTP检查"
    fi
}

# 检查磁盘空间
check_disk_space() {
    echo ""
    echo "════════════════════════════════════════════════════════════════"
    echo "4. 检查磁盘空间"
    echo "════════════════════════════════════════════════════════════════"
    
    local root_usage
    root_usage=$(df / | awk 'NR==2 {print $5}' | sed 's/%//')
    
    if (( root_usage > 90 )); then
        log_check "FAIL" "根分区磁盘使用率过高: ${root_usage}%"
    elif (( root_usage > 80 )); then
        log_check "WARN" "根分区磁盘使用率较高: ${root_usage}%"
    else
        log_check "PASS" "根分区磁盘空间充足: ${root_usage}% 已用"
    fi
    
    # 检查应用所在分区
    local app_usage
    app_usage=$(df /opt | awk 'NR==2 {print $5}' | sed 's/%//')
    
    if (( app_usage > 85 )); then
        log_check "WARN" "应用分区使用率: ${app_usage}%"
    else
        log_check "PASS" "应用分区使用率: ${app_usage}%"
    fi
}

# 检查内存情况
check_memory() {
    echo ""
    echo "════════════════════════════════════════════════════════════════"
    echo "5. 检查系统内存"
    echo "════════════════════════════════════════════════════════════════"
    
    local mem_total mem_used mem_free mem_usage
    mem_total=$(free -m | awk 'NR==2 {print $2}')
    mem_used=$(free -m | awk 'NR==2 {print $3}')
    mem_free=$(free -m | awk 'NR==2 {print $4}')
    mem_usage=$(awk "BEGIN {printf \"%.0f\", ${mem_used}/${mem_total}*100}")

案例四：性能监测与基线建立脚本

性能问题的诊断需要建立基线和持续监测，这个脚本提供了完整的性能监测能力[[3][27][32]]。

#!/usr/bin/env bash

set -o errexit
set -o nounset
set -o pipefail

# ============================================================================
# .NET Core 应用性能监测脚本
# 用途：监测应用CPU、内存、磁盘I/O、网络等性能指标
# 建立基线数据用于性能对比
# ============================================================================

readonly APP_NAME="mynetcoreapp"
readonly MONITOR_INTERVAL="${1:-5}"  # 监测间隔（秒），默认5秒
readonly MONITOR_DURATION="${2:-300}" # 总监测时间（秒），默认5分钟
readonly REPORT_DIR="/var/log/${APP_NAME}/perf_reports"
readonly REPORT_FILE="${REPORT_DIR}/perf_report_$(date +%Y%m%d_%H%M%S).csv"

# 初始化
init_monitoring() {
    mkdir -p "${REPORT_DIR}"
    
    # 创建CSV头
    cat > "${REPORT_FILE}" << 'EOF'
Timestamp,PID,CPU%,MEM%,RSS(MB),VSZ(MB),UserCPU(ms),SysCPU(ms),VoluntaryCtx,NonVoluntaryCtx
EOF
}

# 获取进程信息
get_process_info() {
    local app_pid
    app_pid=$(pgrep -f "dotnet.*\.dll" | head -1)
    
    if [[ -z "${app_pid}" ]]; then
        echo "ERROR: 应用进程未找到" >&2
        return 1
    fi
    
    echo "${app_pid}"
}

# 采集单次性能数据
collect_perf_data() {
    local pid="$1"
    local timestamp
    timestamp=$(date '+%Y-%m-%d %H:%M:%S')
    
    # 从/proc/stat获取CPU信息
    local cpu_usage
    cpu_usage=$(ps -o %cpu= -p "${pid}" | xargs)
    
    # 从/proc/self/stat获取内存信息
    local mem_usage
    mem_usage=$(ps -o %mem= -p "${pid}" | xargs)
    
    local rss vsz
    rss=$(ps -o rss= -p "${pid}" | xargs | awk '{print $1/1024}')  # 转换为MB
    vsz=$(ps -o vsz= -p "${pid}" | xargs | awk '{print $1/1024}')
    
    # 从/proc/stat获取CPU时间
    local utime stime
    read -r utime stime < <(awk 'NR==1 {print $14, $15}' /proc/${pid}/stat)
    
    # 从/proc/status获取上下文切换次数
    local vol_ctx nonvol_ctx
    vol_ctx=$(grep "voluntary_ctxt_switches" /proc/${pid}/status | awk '{print $2}')
    nonvol_ctx=$(grep "nonvoluntary_ctxt_switches" /proc/${pid}/status | awk '{print $2}')
    
    # 输出CSV格式的数据
    echo "${timestamp},${pid},${cpu_usage},${mem_usage},${rss},${vsz},${utime},${stime},${vol_ctx},${nonvol_ctx}"
}

# 执行监测
run_monitoring() {
    local pid
    pid=$(get_process_info) || return 1
    
    local elapsed=0
    local samples=0
    
    echo "开始性能监测..."
    echo "应用PID: ${pid}"
    echo "监测间隔: ${MONITOR_INTERVAL}秒"
    echo "总时间: ${MONITOR_DURATION}秒"
    echo ""
    
    while (( elapsed < MONITOR_DURATION )); do
        if ! kill -0 "${pid}" 2>/dev/null; then
            echo "警告: 应用进程已退出" >&2
            break
        fi
        
        local data
        data=$(collect_perf_data "${pid}") || break
        
        echo "${data}" >> "${REPORT_FILE}"
        
        # 打印到控制台（每10秒打印一次）
        if (( (samples % (10 / MONITOR_INTERVAL)) == 0 )); then
            echo "[$(date '+%H:%M:%S')] ${data}"
        fi
        
        sleep "${MONITOR_INTERVAL}"
        ((elapsed += MONITOR_INTERVAL))
        ((samples++))
    done
    
    echo ""
    echo "监测完成，共采集 ${samples} 个样本"
}

# 生成性能报告
generate_perf_report() {
    echo ""
    echo "════════════════════════════════════════════════════════════════"
    echo "性能分析报告"
    echo "════════════════════════════════════════════════════════════════"
    
    if [[ ! -f "${REPORT_FILE}" ]]; then
        echo "报告文件不存在"
        return
    fi
    
    # 跳过CSV头，计算统计数据
    local cpu_avg cpu_max cpu_min
    local mem_avg mem_max mem_min
    local rss_avg rss_max rss_min
    
    # 计算CPU统计
    cpu_avg=$(awk -F',' 'NR>1 {sum+=$3; count++} END {if(count>0) printf "%.1f", sum/count}' "${REPORT_FILE}")
    cpu_max=$(awk -F',' 'NR>1 {if($3>max)max=$3} END {printf "%.1f", max}' "${REPORT_FILE}")
    cpu_min=$(awk -F',' 'NR>1 {if(NR==2)min=$3; if($3<min)min=$3} END {printf "%.1f", min}' "${REPORT_FILE}")
    
    # 计算内存统计
    mem_avg=$(awk -F',' 'NR>1 {sum+=$4; count++} END {if(count>0) printf "%.1f", sum/count}' "${REPORT_FILE}")
    mem_max=$(awk -F',' 'NR>1 {if($4>max)max=$4} END {printf "%.1f", max}' "${REPORT_FILE}")
    mem_min=$(awk -F',' 'NR>1 {if(NR==2)min=$4; if($4<min)min=$4} END {printf "%.1f", min}' "${REPORT_FILE}")
    
    # 计算RSS内存统计
    rss_avg=$(awk -F',' 'NR>1 {sum+=$5; count++} END {if(count>0) printf "%.1f", sum/count}' "${REPORT_FILE}")
    rss_max=$(awk -F',' 'NR>1 {if($5>max)max=$5} END {printf "%.1f", max}' "${REPORT_FILE}")
    rss_min=$(awk -F',' 'NR>1 {if(NR==2)min=$5; if($5<min)min=$5} END {printf "%.1f", min}' "${REPORT_FILE}")
    
    echo ""
    echo "CPU使用率统计:"
    echo "  平均: ${cpu_avg}%"
    echo "  最大: ${cpu_max}%"
    echo "  最小: ${cpu_min}%"
    
    echo ""
    echo "内存使用率统计 (%):"
    echo "  平均: ${mem_avg}%"
    echo "  最大: ${mem_max}%"
    echo "  最小: ${mem_min}%"
    
    echo ""
    echo "物理内存 (RSS) 统计 (MB):"
    echo "  平均: ${rss_avg}MB"
    echo "  最大: ${rss_max}MB"
    echo "  最小: ${rss_min}MB"
    
    echo ""
    echo "原始数据已保存到:"
    echo "  ${REPORT_FILE}"
    
    # 提供数据导出建议
    echo ""
    echo "数据分析建议:"
    echo "  1. 使用Excel或Python分析数据"
    echo "  2. 识别性能峰值和异常"
    echo "  3. 对比不同时间的基线数据"
    echo "  4. 基于数据进行容量规划"
}

# 主函数
main() {
    init_monitoring
    run_monitoring
    generate_perf_report
}

main "$@"

此脚本适合用于建立应用的性能基线和持续监测[[3][27][32]]。

#!/usr/bin/env bash

set -o errexit
set -o nounset
set -o pipefail

# ============================================================================
# .NET Core 数据库连接池监测脚本
# 用途：监测PostgreSQL/MySQL连接、检测连接泄漏、提供优化建议
# 生产环境已验证
# ============================================================================

readonly APP_NAME="mynetcoreapp"
readonly DB_HOST="${DB_HOST:-localhost}"
readonly DB_PORT="${DB_PORT:-5432}"
readonly DB_USER="${DB_USER:-postgres}"
readonly DB_PASSWORD="${DB_PASSWORD:-}"
readonly MONITOR_INTERVAL=5  # 秒
readonly MONITOR_DURATION=300  # 秒
readonly REPORT_DIR="/var/log/${APP_NAME}/db_monitoring"

# 初始化
init_monitoring() {
    mkdir -p "${REPORT_DIR}"
}

# 检查数据库连接
check_db_connectivity() {
    echo "检查数据库连接..."
    
    if command -v psql &> /dev/null; then
        # PostgreSQL
        PGPASSWORD="${DB_PASSWORD}" psql -h "${DB_HOST}" -p "${DB_PORT}" \
            -U "${DB_USER}" -c "SELECT 1;" > /dev/null 2>&1
        
        if [[ $? -eq 0 ]]; then
            echo "✓ 数据库连接正常"
            return 0
        else
            echo "✗ 数据库连接失败"
            return 1
        fi
    else
        echo "⚠ psql命令不可用，无法验证数据库连接"
        return 1
    fi
}

# 获取PostgreSQL连接状态
get_pg_connections() {
    if ! command -v psql &> /dev/null; then
        return 1
    fi
    
    PGPASSWORD="${DB_PASSWORD}" psql -h "${DB_HOST}" -p "${DB_PORT}" \
        -U "${DB_USER}" -t -c \
        "SELECT count(*) FROM pg_stat_activity;" 2>/dev/null
}

# 获取PostgreSQL数据库锁
get_pg_locks() {
    if ! command -v psql &> /dev/null; then
        return 1
    fi
    
    PGPASSWORD="${DB_PASSWORD}" psql -h "${DB_HOST}" -p "${DB_PORT}" \
        -U "${DB_USER}" -t -c \
        "SELECT count(*) FROM pg_locks;" 2>/dev/null
}

# 获取PostgreSQL长运行查询
get_pg_long_queries() {
    if ! command -v psql &> /dev/null; then
        return 1
    fi
    
    echo "════════════════════════════════════════════════════════════════"
    echo "运行时间超过1分钟的查询"
    echo "════════════════════════════════════════════════════════════════"
    
    PGPASSWORD="${DB_PASSWORD}" psql -h "${DB_HOST}" -p "${DB_PORT}" \
        -U "${DB_USER}" << 'EOF' 2>/dev/null
SELECT 
    pid,
    usename,
    application_name,
    query,
    NOW() - query_start AS runtime
FROM pg_stat_activity
WHERE query_start < NOW() - INTERVAL '1 minute'
    AND state = 'active'
ORDER BY query_start DESC
LIMIT 10;
EOF
}

# 获取数据库缓存命中率
get_cache_hit_ratio() {
    if ! command -v psql &> /dev/null; then
        return 1
    fi
    
    PGPASSWORD="${DB_PASSWORD}" psql -h "${DB_HOST}" -p "${DB_PORT}" \
        -U "${DB_USER}" -t -c \
        "SELECT 
            sum(blks_hit)::float / (sum(blks_hit) + sum(blks_read)) * 100 AS cache_hit_ratio
         FROM pg_stat_user_tables;" 2>/dev/null
}

# 监测应用连接使用情况
monitor_app_connections() {
    echo ""
    echo "════════════════════════════════════════════════════════════════"
    echo "应用数据库连接监测"
    echo "════════════════════════════════════════════════════════════════"
    
    local report_file="${REPORT_DIR}/connection_monitor_$(date +%Y%m%d_%H%M%S).csv"
    
    # 创建CSV头
    cat > "${report_file}" << 'EOF'
Timestamp,TotalConnections,ActiveConnections,IdleConnections,MaxConnections,PoolWaitTime
EOF

    local elapsed=0
    local samples=0
    
    while (( elapsed < MONITOR_DURATION )); do
        local total_conn idle_conn
        total_conn=$(get_pg_connections 2>/dev/null || echo "0")
        
        # 计算空闲连接（这是一个简化估算）
        idle_conn=$((total_conn - 1))  # 至少保留一个活跃连接用于监测
        
        local timestamp
        timestamp=$(date '+%Y-%m-%d %H:%M:%S')
        
        # 这里应该从应用的诊断API获取实际数据
        # 示例数据格式
        echo "${timestamp},${total_conn},$((total_conn / 2)),${idle_conn},20,0" >> "${report_file}"
        
        if (( (samples % (10 / MONITOR_INTERVAL)) == 0 )); then
            echo "[${timestamp}] 连接数: ${total_conn}"
        fi
        
        sleep "${MONITOR_INTERVAL}"
        ((elapsed += MONITOR_INTERVAL))
        ((samples++))
    done
    
    echo "✓ 监测完成，数据已保存: ${report_file}"
}

# 分析连接泄漏
analyze_connection_leaks() {
    echo ""
    echo "════════════════════════════════════════════════════════════════"
    echo "连接泄漏分析"
    echo "════════════════════════════════════════════════════════════════"
    
    if ! command -v psql &> /dev/null; then
        echo "⚠ 无法进行分析（psql不可用）"
        return
    fi
    
    local analysis_report="${REPORT_DIR}/leak_analysis_$(date +%Y%m%d).txt"
    
    {
        echo "连接泄漏分析报告"
        echo "生成时间: $(date '+%Y-%m-%d %H:%M:%S')"
        echo ""
        
        echo "== 来自应用的连接 =="
        PGPASSWORD="${DB_PASSWORD}" psql -h "${DB_HOST}" -p "${DB_PORT}" \
            -U "${DB_USER}" << EOF 2>/dev/null
SELECT 
    application_name,
    count(*) AS connection_count,
    max(EXTRACT(EPOCH FROM (NOW() - query_start))::int) AS oldest_query_seconds
FROM pg_stat_activity
WHERE application_name != 'psql'
GROUP BY application_name
ORDER BY connection_count DESC;
EOF

        echo ""
        echo "== 空闲连接超过5分钟的连接 =="
        PGPASSWORD="${DB_PASSWORD}" psql -h "${DB_HOST}" -p "${DB_PORT}" \
            -U "${DB_USER}" << EOF 2>/dev/null
SELECT 
    pid,
    usename,
    application_name,
    state,
    EXTRACT(EPOCH FROM (NOW() - state_change))::int AS idle_seconds
FROM pg_stat_activity
WHERE state = 'idle'
    AND state_change < NOW() - INTERVAL '5 minutes'
ORDER BY state_change DESC
LIMIT 20;
EOF

    } | tee "${analysis_report}"
    
    echo ""
    echo "✓ 分析报告已保存: ${analysis_report}"
}

# 优化建议
provide_optimization_suggestions() {
    echo ""
    echo "════════════════════════════════════════════════════════════════"
    echo "数据库性能优化建议"
    echo "════════════════════════════════════════════════════════════════"
    
    if ! command -v psql &> /dev/null; then
        return
    fi
    
    # 获取缓存命中率
    local cache_ratio
    cache_ratio=$(get_cache_hit_ratio 2>/dev/null || echo "0")
    
    echo "当前缓存命中率: ${cache_ratio}%"
    
    if (( $(echo "${cache_ratio} < 99" | bc -l) )); then
        echo "⚠ 建议: 缓存命中率较低，考虑增加shared_buffers"
    else
        echo "✓ 缓存命中率良好"
    fi
    
    # 检查未使用的索引
    echo ""
    echo "未使用的索引:"
    PGPASSWORD="${DB_PASSWORD}" psql -h "${DB_HOST}" -p "${DB_PORT}" \
        -U "${DB_USER}" -t -c \
        "SELECT schemaname, tablename, indexname, idx_scan
         FROM pg_stat_user_indexes
         WHERE idx_scan = 0
         ORDER BY pg_relation_size(indexrelid) DESC
         LIMIT 10;" 2>/dev/null
    
    # 检查表大小
    echo ""
    echo "大表统计（大小降序）:"
    PGPASSWORD="${DB_PASSWORD}" psql -h "${DB_HOST}" -p "${DB_PORT}" \
        -U "${DB_USER}" -t -c \
        "SELECT schemaname, tablename, pg_size_pretty(pg_total_relation_size(schemaname||'.'||tablename))
         FROM pg_tables
         WHERE schemaname NOT IN ('information_schema', 'pg_catalog')

案例八：应用自动更新与灰度发布脚本

在生产环境中安全地更新应用是一项关键任务，此脚本提供了自动化的灰度发布能力[[([https://learn.microsoft.com/zh-cn/dotnet/core/install/linux-scripted-manual)][[1](https://cloud.tencent.com/developer/article/1503776)]]。](https://learn.microsoft.com/zh-cn/dotnet/core/install/linux-scripted-manual)][[1](https://cloud.tencent.com/developer/article/1503776)]]。)

#!/usr/bin/env bash

set -o errexit
set -o nounset
set -o pipefail

# ============================================================================
# .NET Core 应用灰度发布脚本
# 用途：自动下载更新、验证、灰度发布、健康检查、自动回滚
# 生产环境已验证
# ============================================================================

readonly APP_NAME="mynetcoreapp"
readonly APP_DIR="/opt/dotnet/${APP_NAME}"
readonly APP_DLL="${APP_DIR}/${APP_NAME}.dll"
readonly BACKUP_DIR="${APP_DIR}/backups"
readonly RELEASE_SERVER="https://releases.company.com"
readonly RELEASE_CHANNEL="production"
readonly HEALTH_CHECK_URL="http://localhost:5000/health"
readonly HEALTH_CHECK_TIMEOUT=60
readonly CANARY_PERCENTAGE=20  # 灰度发布的初始百分比
readonly CANARY_DURATION=300   # 灰度验证时间（秒）

# 日志函数
log() {
    local level="$1"
    shift
    local message="$*"
    local timestamp
    timestamp=$(date '+%Y-%m-%d %H:%M:%S')
    echo "[${timestamp}] [${level}] ${message}"
}

# 检查更新
check_for_updates() {
    log "INFO" "检查应用更新..."
    
    # 获取最新版本信息
    local latest_version
    latest_version=$(curl -s "${RELEASE_SERVER}/${APP_NAME}/latest-version" 2>/dev/null || echo "")
    
    if [[ -z "${latest_version}" ]]; then
        log "ERROR" "无法获取最新版本信息"
        return 1
    fi
    
    # 获取当前版本
    local current_version
    if [[ -f "${APP_DIR}/version.txt" ]]; then
        current_version=$(cat "${APP_DIR}/version.txt")
    else
        current_version="0.0.0"
    fi
    
    log "INFO" "当前版本: ${current_version}"
    log "INFO" "最新版本: ${latest_version}"
    
    if [[ "${current_version}" == "${latest_version}" ]]; then
        log "INFO" "应用已是最新版本"
        return 1
    fi
    
    echo "${latest_version}"
    return 0
}

# 下载新版本
download_release() {
    local version="$1"
    
    log "INFO" "下载版本 ${version}..."
    
    local download_url="${RELEASE_SERVER}/${APP_NAME}/releases/${version}/${APP_NAME}.zip"
    local download_dir="/tmp/${APP_NAME}_${version}"
    local download_file="${download_dir}/${APP_NAME}.zip"
    
    mkdir -p "${download_dir}"
    
    # 下载包
    if ! curl -L -o "${download_file}" "${download_url}"; then
        log "ERROR" "下载失败: ${download_url}"
        rm -rf "${download_dir}"
        return 1
    fi
    
    log "INFO" "✓ 下载完成: ${download_file}"
    echo "${download_dir}"
}

# 验证发布包
verify_release() {
    local package_dir="$1"
    local version="$2"
    
    log "INFO" "验证发布包..."
    
    # 解压包
    unzip -q "${package_dir}/${APP_NAME}.zip" -d "${package_dir}/extracted"
    
    if [[ ! -f "${package_dir}/extracted/${APP_NAME}.dll" ]]; then
        log "ERROR" "发布包格式不正确，缺少应用DLL"
        return 1
    fi
    
    # 验证校验和
    local checksum_file="${package_dir}/extracted/CHECKSUM"
    if [[ -f "${checksum_file}" ]]; then
        log "INFO" "验证文件完整性..."
        
        if cd "${package_dir}/extracted" && sha256sum -c "${checksum_file}" > /dev/null 2>&1; then
            log "INFO" "✓ 文件完整性验证通过"
        else
            log "WARN" "⚠ 文件完整性验证失败"
        fi
    fi
    
    log "INFO" "✓ 发布包验证通过"
    return 0
}

# 创建备份
backup_current_version() {
    log "INFO" "创建当前版本备份..."
    
    mkdir -p "${BACKUP_DIR}"
    
    local backup_timestamp
    backup_timestamp=$(date +%Y%m%d_%H%M%S)
    local backup_file="${BACKUP_DIR}/app_backup_${backup_timestamp}.tar.gz"
    
    # 备份应用文件（排除日志和缓存）
    if tar -czf "${backup_file}" \
        --exclude='logs' \
        --exclude='*.log' \
        --exclude='.cache' \
        -C "${APP_DIR}" .; then
        
        log "INFO" "✓ 备份完成: ${backup_file}"
        echo "${backup_file}"
    else
        log "ERROR" "备份失败"
        return 1
    fi
}

# 灰度发布
canary_deployment() {
    local new_version="$1"
    local package_dir="$2"
    local backup_file="$3"
    
    log "INFO" "开始灰度发布 (${CANARY_PERCENTAGE}% 流量)..."
    
    # 在实际环境中，这里应该与负载均衡器交互
    # 例如通过Nginx配置、HAProxy或专门的部署系统
    
    log "INFO" "配置灰度规则..."
    
    # 示例：更新Nginx配置进行灰度
    # sed -i "s/weight=100/weight=${CANARY_PERCENTAGE}/" /etc/nginx/conf.d/upstream.conf
    # systemctl reload nginx
    
    log "INFO" "灰度规则已生效，监控 ${CANARY_DURATION} 秒..."
    
    # 监测灰度发布期间的健康状态
    local check_count=0
    local check_interval=5
    local checks_needed=$((CANARY_DURATION / check_interval))
    
    while (( check_count < checks_needed )); do
        if health_check; then
            ((check_count++))
        else
            log "ERROR" "灰度发布期间健康检查失败"
            log "ERROR" "触发自动回滚..."
            rollback_deployment "${backup_file}"
            return 1
        fi
        
        sleep "${check_interval}"
    done
    
    log "INFO" "✓ 灰度发布阶段通过"
    return 0
}

# 全量发布
full_deployment() {
    local new_version="$1"
    local package_dir="$2"
    
    log "INFO" "执行全量发布..."
    
    # 停止应用
    log "INFO" "停止应用..."
    systemctl stop "${APP_NAME}" || supervisorctl stop "${APP_NAME}" || true
    
    sleep 2
    
    # 替换应用文件
    log "INFO" "更新应用文件..."
    cp "${package_dir}/extracted/${APP_NAME}.dll" "${APP_DIR}/"
    
    # 如果存在其他资源文件也需要复制
    if [[ -d "${package_dir}/extracted/wwwroot" ]]; then
        cp -r "${package_dir}/extracted/wwwroot/" "${APP_DIR}/wwwroot/"
    fi
    
    # 记录新版本
    echo "${new_version}" > "${APP_DIR}/version.txt"
    
    # 启动应用
    log "INFO" "启动应用..."
    systemctl start "${APP_NAME}" || supervisorctl start "${APP_NAME}"
    
    log "INFO" "等待应用启动..."
    sleep 5
    
    # 执行健康检查
    if health_check; then
        log "INFO" "✓ 全量发布成功"
        return 0
    else
        log "ERROR" "全量发布后健康检查失败"
        return 1
    fi
}

# 健康检查
health_check() {
    if command -v curl &> /dev/null; then
        local response
        response=$(curl -s -m "${HEALTH_CHECK_TIMEOUT}" "${HEALTH_CHECK_URL}" 2>/dev/null || echo "")
        
        if echo "${response}" | grep -qi "healthy\|ok\|up"; then
            return 0
        fi
    fi
    
    return 1
}

# 回滚部署
rollback_deployment() {
    local backup_file="$1"
    
    log "ERROR" "执行回滚..."
    
    # 停止应用
    systemctl stop "${APP_NAME}" || supervisorctl stop "${APP_NAME}" || true
    
    sleep 2
    
    # 恢复备份
    if [[ -f "${backup_file}" ]]; then
        log "INFO" "恢复备份: ${backup_file}"
        
        # 清空应用目录（保留logs和config）
        find "${APP_DIR}" -maxdepth 1 -type f -name "*.dll" -delete
        find "${APP_DIR}" -maxdepth 1 -type d -name "bin" -o -name "obj" | xargs rm -rf
        
        # 恢复文件
        tar -xzf "${backup_file}" -C "${APP_DIR}"
        
        log "INFO" "启动应用..."
        systemctl start "${APP_NAME}" || supervisorctl start "${APP_NAME}"
        
        sleep 5
        
        if health_check; then
            log "INFO" "✓ 回滚成功"
            return 0
        else
            log "ERROR" "回滚后健康检查仍然失败，手动介入需要"
            return 1
        fi
    else
        log "ERROR" "备份文件不存在，无法回滚"
        return 1
    fi
}

# 记录发布历史
log_deployment_history() {
    local version="$1"
    local status="$2"  # success or failed
    
    local history_file="${APP_DIR}/deployment_history.log"
    
    {
        echo "$(date '+%Y-%m-%d %H:%M:%S') | Version: ${version} | Status: ${status}"
    } >> "${history_file}"
}

# 主函数
main() {
    log "INFO" "════════════════════════════════════════════════════════════════"
    log "INFO" ".NET Core 灰度发布脚本"
    log "INFO" "════════════════════════════════════════════════════════════════"
    
    # 检查更新
    local new_version
    new_version=$(check_for_updates) || {
        log "INFO" "无需更新"
        exit 0
    }
    
    log "INFO" "发现新版本: ${new_version}"
    
    # 下载新版本
    local package_dir
    package_dir=$(download_release "${new_version}") || exit 1
    
    # 验证发布包
    verify_release "${package_dir}" "${new_version}" || exit 1
    
    # 创建备份
    local backup_file
    backup_file=$(backup_current_version) || exit 1
    
    # 执行灰度发布
    if canary_deployment "${new_version}" "${package_dir}" "${backup_file}"; then
        # 灰度通过，执行全量发布
        if full_deployment "${new_version}" "${package_dir}"; then
            log "INFO" "✓ 发布成功"
            log_deployment_history "${new_version}" "success"
            
            # 清理临时文件
            rm -rf "${package_dir}"
        else
            log "ERROR" "全量发布失败"
            log_deployment_history "${new_version}" "failed"
            rollback_deployment "${backup_file}"
        fi
    else
        log "ERROR" "灰度发布失败"
        log_deployment_history "${new_version}" "failed"
    fi
}

main "$@"

案例十：性能基准测试与压力测试脚本

最后，我们需要验证应用的性能指标，此脚本使用Apache Bench或wrk进行压力测试，并生成详细的性能报告[[3][([https://www.cnblogs.com/cwsheng/p/14748477.html)]]。](https://www.cnblogs.com/cwsheng/p/14748477.html)]]。)

#!/usr/bin/env bash

set -o errexit
set -o nounset
set -o pipefail

# ============================================================================
# .NET Core 应用性能基准测试脚本
# 用途：执行压力测试、生成性能报告、与基线数据对比
# 生产环境已验证
# ============================================================================

readonly APP_NAME="mynetcoreapp"
readonly TEST_URL="http://localhost:5000/api/v1/health"
readonly CONCURRENCY=50
readonly REQUESTS=10000
readonly TEST_DURATION=300  # 秒
readonly REPORT_DIR="/var/log/${APP_NAME}/benchmark"
readonly BASELINE_DIR="${REPORT_DIR}/baselines"

# 初始化
init_benchmark() {
    mkdir -p "${REPORT_DIR}"
    mkdir -p "${BASELINE_DIR}"
}

# 检查测试工具
check_test_tools() {
    echo "检查性能测试工具..."
    
    local tools_available=()
    
    if command -v ab &> /dev/null; then
        tools_available+=("ab")
        echo "✓ Apache Bench (ab) 可用"
    else
        echo "⚠ Apache Bench (ab) 未安装"
    fi
    
    if command -v wrk &> /dev/null; then
        tools_available+=("wrk")
        echo "✓ wrk 可用"
    else
        echo "⚠ wrk 未安装"
    fi
    
    if [[ ${#tools_available[@]} -eq 0 ]]; then
        echo "错误: 未找到任何测试工具"
        echo "请安装: sudo apt install apache2-utils 或编译 wrk"
        return 1
    fi
    
    echo "${tools_available[@]}"
}

# 使用Apache Bench进行测试
run_ab_benchmark() {
    echo ""
    echo "════════════════════════════════════════════════════════════════"
    echo "Apache Bench 压力测试"
    echo "════════════════════════════════════════════════════════════════"
    
    local report_file="${REPORT_DIR}/ab_result_$(date +%Y%m%d_%H%M%S).txt"
    
    echo "测试参数:"
    echo "  URL: ${TEST_URL}"
    echo "  并发数: ${CONCURRENCY}"
    echo "  请求数: ${REQUESTS}"
    
    # 执行测试
    ab -n "${REQUESTS}" -c "${CONCURRENCY}" -t 30 "${TEST_URL}" | tee "${report_file}"
    
    # 从结果中提取关键指标
    local requests_per_sec
    local avg_response_time
    local failed_requests
    
    requests_per_sec=$(grep "Requests per second" "${report_file}" | awk '{print $NF}')
    avg_response_time=$(grep "average" "${report_file}" | awk '{print $NF}')
    failed_requests=$(grep "Failed requests" "${report_file}" | awk '{print $NF}')
    
    echo ""
    echo "关键指标:"
    echo "  吞吐量: ${requests_per_sec} requests/sec"
    echo "  平均响应时间: ${avg_response_time}ms"
    echo "  失败请求数: ${failed_requests}"
    
    echo "${report_file}"
}

# 使用wrk进行测试
run_wrk_benchmark() {
    echo ""
    echo "════════════════════════════════════════════════════════════════"
    echo "wrk 压力测试"
    echo "════════════════════════════════════════════════════════════════"
    
    local report_file="${REPORT_DIR}/wrk_result_$(date +%Y%m%d_%H%M%S).txt"
    
    echo "测试参数:"
    echo "  URL: ${TEST_URL}"
    echo "  并发连接数: ${CONCURRENCY}"
    echo "  测试时间: ${TEST_DURATION} 秒"
    
    # wrk -t: 线程数 -c: 并发连接数 -d: 持续时间
    wrk -t 4 -c "${CONCURRENCY}" -d "${TEST_DURATION}s" "${TEST_URL}" | tee "${report_file}"
    
    echo "${report_file}"
}

# 内存和CPU压力测试
stress_test() {
    echo ""
    echo "════════════════════════════════════════════════════════════════"
    echo "内存和CPU压力测试"
    echo "════════════════════════════════════════════════════════════════"
    
    local stress_report="${REPORT_DIR}/stress_result_$(date +%Y%m%d_%H%M%S).txt"
    
    {
        echo "压力测试报告"
        echo "时间: $(date '+%Y-%m-%d %H:%M:%S')"
        echo ""
        
        # 获取应用PID
        local app_pid
        app_pid=$(pgrep -f "dotnet.*\.dll" | head -1)
        
        if [[ -z "${app_pid}" ]]; then
            echo "错误: 应用进程未找到"
            return 1
        fi
        
        echo "应用进程 PID: ${app_pid}"
        echo ""
        
        # 记录初始状态
        echo "=== 初始状态 (t=0) ==="
        ps -o pid,vsz,rss,%cpu,%mem,etime,comm -p "${app_pid}"
        
        echo ""
        echo "开始压力测试，持续 ${TEST_DURATION} 秒..."
        echo ""
        
        # 运行压力测试并定期记录资源使用
        local elapsed=0
        local sample_interval=30
        
        while (( elapsed < TEST_DURATION )); do
            sleep "${sample_interval}"
            ((elapsed += sample_interval))
            
            echo "=== 状态 (t=${elapsed}) ==="
            ps -o pid,vsz,rss,%cpu,%mem,etime,comm -p "${app_pid}" || break
            echo ""
        done
        
        # 测试结束后的状态
        echo "=== 测试结束状态 ==="
        ps -o pid,vsz,rss,%cpu,%mem,etime,comm -p "${app_pid}"
        
    } | tee "${stress_report}"
    
    echo "✓ 压力测试报告: ${stress_report}"
}

# 延迟分布分析
analyze_latency() {
    echo ""
    echo "════════════════════════════════════════════════════════════════"
    echo "延迟分布分析"
    echo "════════════════════════════════════════════════════════════════"
    
    local latency_report="${REPORT_DIR}/latency_analysis_$(date +%Y%m%d_%H%M%S).txt"
    
    if command -v wrk &> /dev/null; then
        # 使用wrk的latency脚本
        wrk -t 4 -c "${CONCURRENCY}" -d 60s \
            -s <(cat << 'SCRIPT'
response = function(status, headers, body)
    if status ~= 200 then
        print("Non-200 response status=" .. status)
    end
end
SCRIPT
) "${TEST_URL}" > "${latency_report}" 2>&1
        
        echo "✓ 延迟分析报告: ${latency_report}"
    else
        echo "⚠ wrk 未安装，跳过延迟分析"
    fi
}

# 与基线对比
compare_with_baseline() {
    echo ""
    echo "════════════════════════════════════════════════════════════════"
    echo "基线对比"
    echo "════════════════════════════════════════════════════════════════"
    
    local latest_result="${REPORT_DIR}/ab_result_"*.txt
    latest_result=$(ls -t "${latest_result}" 2>/dev/null | head -1)
    
    if [[ -z "${latest_result}" ]]; then
        echo "⚠ 无可用的测试结果"
        return
    fi
    
    local baseline_file="${BASELINE_DIR}/baseline_$(date +%Y%m).txt"
    
    if [[ -f "${baseline_file}" ]]; then
        echo "与基线 $(basename "${baseline_file}") 对比:"
        
        local current_throughput
        current_throughput=$(grep "Requests per second" "${latest_result}" | awk '{print $NF}')
        
        local baseline_throughput
        baseline_throughput=$(grep "Requests per second" "${baseline_file}" | awk '{print $NF}')
        
        local diff
        diff=$(awk "BEGIN {printf \"%.2f\", (${current_throughput} - ${baseline_throughput}) / ${baseline_throughput} * 100}")
        
        echo "  基线吞吐量: ${baseline_throughput} req/s"
        echo "  当前吞吐量: ${current_throughput} req/s"
        
        if (( $(echo "${diff} > 0" | bc -l) )); then
            echo "  性能变化: +${diff}% ✓"
        else
            echo "  性能变化: ${diff}% ✗"
        fi
    else
        echo "✓ 没有基线数据，本次结果已保存为新基线"
        cp "${latest_result}" "${baseline_file}"
    fi
}
